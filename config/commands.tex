%% ============================================================
%% COMMANDS — Reusable custom commands and environments
%% ============================================================

%% ----- Placeholder text (gray italics — remove in final version) -----
\newcommand{\placeholder}[1]{\textit{\textcolor{placeholdergray}{#1}}}

%% ----- Unnumbered section that appears in TOC -----
\newcommand{\usection}[1]{%
  \clearpage
  \phantomsection
  \section*{\MakeUppercase{#1}}%
  \addcontentsline{toc}{section}{#1}%
}

%% ----- Signature block -----
%% Usage: \sigblock{Name}{Designation}{Department}{University}
\newcommand{\sigblock}[4]{%
  \vspace{1.5cm}%
  \noindent\makebox[3in]{\dotfill}\\[2pt]
  \noindent\textbf{#1}\\
  #2\\
  #3\\
  #4%
}

%% ----- Compact signature (for multiple on one page) -----
\newcommand{\sigblockcompact}[4]{%
  \vspace{1cm}%
  \noindent\makebox[3in]{\dotfill}\\[2pt]
  \noindent\textbf{#1}\\
  #2\\
  #3\\
  #4%
}

%% ----- Right-aligned role label -----
\newcommand{\rolelabel}[1]{%
  \hfill\textbf{#1}\par%
}

%% ----- Gantt cell (shaded) -----
\newcommand{\gantt}{\cellcolor{ganttblue}}

%% ----- Empty Gantt cell -----
\newcommand{\nogant}{}
